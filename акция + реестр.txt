def write_df_preserve_format(ws, df):
    """
    Записывает значения ТОЛЬКО в существующие колонки Excel.
    Старое форматирование сохраняется.
    Новые колонки (которых нет в Excel) НЕ трогает.
    """
    # карта заголовков Excel
    header_map = {
        str(cell.value).strip(): idx + 1
        for idx, cell in enumerate(ws[1])
        if cell.value is not None
    }

    # чистим ТОЛЬКО существующие колонки
    for col_name, col_idx in header_map.items():
        if col_name not in df.columns:
            continue
        for row in ws.iter_rows(min_row=2, min_col=col_idx, max_col=col_idx):
            row[0].value = None

    # пишем данные ТОЛЬКО в совпадающие заголовки
    for r_idx, (_, row) in enumerate(df.iterrows(), start=2):
        for col_name, value in row.items():
            if col_name not in header_map:
                # новая колонка — пропускаем (будет без формата)
                continue
            ws.cell(row=r_idx, column=header_map[col_name]).value = (
                None if pd.isna(value) else value
            )
def remove_excel_tables(ws):
    """
    Полностью удаляет Excel-таблицы (ListObject) с листа,
    превращая их в обычный диапазон.
    Формулы, данные, форматирование сохраняются.
    """
    if hasattr(ws, "tables") and ws.tables:
        ws.tables.clear()

def normalize_dog_key(s: pd.Series) -> pd.Series:
    return (
        s.astype(str)
        .str.replace("\u200b", "", regex=False)
        .str.replace("\xa0", "", regex=False)
        .str.replace(" ", "", regex=False)
        .str.replace("–", "-", regex=False)
        .str.replace("—", "-", regex=False)
        .str.strip()
    )
def normalize_ipn_key(s: pd.Series) -> pd.Series:
    return (
        s.astype(str)
        .str.replace("\u200b", "", regex=False)
        .str.replace("\xa0", "", regex=False)
        .str.replace(" ", "", regex=False)
        .str.replace(".0", "", regex=False)
        .str.strip()
    )
def clean_header(x):
    return (
        str(x)
        .replace("\u200b", "")
        .replace("\xa0", " ")
        .replace("\n", " ")
        .replace("\r", " ")
        .strip()
        .replace("  ", " ")
    )

def _a1_to_coords(a1: str):
    # "A1" -> ("A", 1)
    m = re.match(r"^([A-Z]+)(\d+)$", a1)
    return m.group(1), int(m.group(2))

def _range_to_bounds(rng: str):
    # "A1:K20" -> (min_col, min_row, max_col, max_row)
    left, right = rng.split(":")
    c1, r1 = _a1_to_coords(left)
    c2, r2 = _a1_to_coords(right)

    def col_to_int(col_letters: str) -> int:
        n = 0
        for ch in col_letters:
            n = n * 26 + (ord(ch) - 64)
        return n

    return col_to_int(c1), r1, col_to_int(c2), r2

def _bounds_to_range(min_col, min_row, max_col, max_row):
    return f"{get_column_letter(min_col)}{min_row}:{get_column_letter(max_col)}{max_row}"

def _extend_tables_after_insert(ws, insert_at_col: int, inserted_cols: int):
    """
    Обновляет диапазоны Excel-таблиц на листе, если вставка колонок попала внутрь таблицы.
    """
    # ws.tables может быть dict-like
    tables = getattr(ws, "tables", None)
    if not tables:
        return

    for name, tbl in ws.tables.items():
        min_c, min_r, max_c, max_r = _range_to_bounds(tbl.ref)

        # если вставка произошла внутри таблицы (или на ее границе слева)
        # то правую границу таблицы надо сдвинуть вправо
        if insert_at_col <= max_c:
            max_c += inserted_cols
            new_ref = _bounds_to_range(min_c, min_r, max_c, max_r)
            tbl.ref = new_ref

            # автофильтр тоже обновим (если есть)
            if getattr(tbl, "autoFilter", None) is not None and getattr(tbl.autoFilter, "ref", None):
                # автофильтр обычно равен ref без заголовков/или совпадает — безопасно выставить как ref
                tbl.autoFilter.ref = new_ref

def ensure_calc_columns_after_count(ws, calc_cols, after_col_name):
    """
    Вставляет расчетные колонки сразу после after_col_name.
    И ВАЖНО: если на листе есть Excel-таблица (ListObject),
    расширяет её диапазон, чтобы Excel не показывал 'восстановление'.
    """
    headers = [cell.value for cell in ws[1]]

    if after_col_name not in headers:
        raise ValueError(f"❌ Колонка '{after_col_name}' не найдена в Excel")

    insert_pos = headers.index(after_col_name) + 2  # 1-based, сразу после after_col_name

    inserted_total = 0
    for col in calc_cols:
        # актуализируем заголовки после каждой вставки
        headers_now = [cell.value for cell in ws[1]]
        if col in headers_now:
            continue

        ws.insert_cols(insert_pos)
        _extend_tables_after_insert(ws, insert_pos, 1)

        ws.cell(row=1, column=insert_pos).value = col
        insert_pos += 1
        inserted_total += 1

def write_df_by_headers(ws, df):
    header_map = {
        clean_header(cell.value): i + 1
        for i, cell in enumerate(ws[1])
        if cell.value is not None
    }

    # ❗ ЧИСТИМ ТОЛЬКО КОЛОНКИ, КОТОРЫЕ БУДЕМ ПИСАТЬ
    for col_name, col_idx in header_map.items():
        if col_name not in df.columns:
            continue
        for row in ws.iter_rows(min_row=2, min_col=col_idx, max_col=col_idx):
            row[0].value = None

    # ✍️ ПИШЕМ ЗНАЧЕНИЯ
    for r_idx, (_, row) in enumerate(df.iterrows(), start=2):
        for col_name, value in row.items():
            col_name_clean = clean_header(col_name)
            if col_name_clean not in header_map:
                continue
            ws.cell(
                row=r_idx,
                column=header_map[col_name_clean]
            ).value = None if pd.isna(value) else value


def norm(x):
    return str(x).strip().lower().replace("\xa0", " ")

def find_col(cols, keywords):
    for c in cols:
        if all(k in norm(c) for k in keywords):
            return c
    raise ValueError(f"❌ Не найдена колонка: {keywords}")


def safe_to_numeric(series):
    try:
        return pd.to_numeric(series)
    except:
        return series

def to_decimal(x):
    if pd.isna(x):
        return Decimal("0")
    s = str(x).replace("\xa0", "").replace(" ", "").replace(",", ".")
    try:
        return Decimal(s)
    except:
        return Decimal("0")

def normalize_number_like_text(s: pd.Series) -> pd.Series:
    return (
        s.astype(str)
        .str.replace("\xa0", " ", regex=False)
        .str.strip()
        .str.replace(r"\.0$", "", regex=True)
    )

def detect_factoring_name(x):
    if pd.isna(x):
        return "Споживчий центр"

    s = str(x).upper()

    if "УКФ" in s:
        return "Укр Кредит Фінанс"
    if "МФЮ" in s:
        return "Маніфою"
    if "НІ" in s:
        return "ТАСКОМБАНК"
    if "SC" in s:
        return "СелфіКредіт"

    return "Споживчий центр"

# 🔒 ДАТЫ: СТРОГОЕ ПРИВЕДЕНИЕ К dd.mm.yyyy БЕЗ ПЕРЕПУТЫВАНИЙ
# ===============================
_re_iso = re.compile(r"^\s*(\d{4})-(\d{2})-(\d{2})(?:\s+\d{1,2}:\d{2}(:\d{2})?)?\s*$")
_re_dmy = re.compile(r"^\s*(\d{2})\.(\d{2})\.(\d{4})(?:\s+\d{1,2}:\d{2}(:\d{2})?)?\s*$")

def normalize_date_str(x) -> str | None:
    if x is None or pd.isna(x):
        return None
    s = str(x).strip()
    if s == "" or s.lower() in ("nan", "none"):
        return None

    # 1) ISO: YYYY-MM-DD [time] -> DD.MM.YYYY
    m = _re_iso.match(s)
    if m:
        yyyy, mm, dd = m.group(1), m.group(2), m.group(3)
        return f"{dd}.{mm}.{yyyy}"

    # 2) DMY: DD.MM.YYYY [time] -> DD.MM.YYYY
    m = _re_dmy.match(s)
    if m:
        dd, mm, yyyy = m.group(1), m.group(2), m.group(3)
        return f"{dd}.{mm}.{yyyy}"

    # 3) иначе не трогаем (чтобы не сломать нестандарт)
    return s

def normalize_date_cols(df: pd.DataFrame) -> pd.DataFrame:
    date_cols = [c for c in df.columns if "дата" in norm(c)]
    for c in date_cols:
        df[c] = df[c].apply(normalize_date_str)
    return df

def category(reg, r):
    if str(reg).strip() in ("УКФ-261224", "ТАС-061224"):
        return "не расчитывать"
    if r is None or pd.isna(r):
        return "проверить"

    if r < 1.5: return "тело +10%"
    if r < 1.8: return "тело +20%"
    if r < 2.1: return "тело +30%"
    if r < 2.4: return "тело +40%"
    if r < 2.7: return "тело +50%"
    if r < 3.0: return "тело +60%"
    if r < 3.3: return "тело +70%"
    if r < 3.7: return "тело +80%"
    return "тело +90%"

def calc_action(r):
    cat = r["категория"]
    dog = r["_DOG_KEY"]

        v = ukf_map.get(dog)
        if v is None or str(v).strip() == "":

        try:
            return int(str(v).replace(" ", "").replace("\xa0", "").replace(",", ".")) if v is not None else None
        except:
            return None

    if cat == "проверить":
        return None

    return int(r["_body"] * Decimal(str(COEF[cat])))
